# i18n 国际化支持 Implementation Plan

> **For agentic workers:** REQUIRED SUB- SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 建立最小可用的国际化（i18n）层，将玩家可见中文文本从业务代码迁移到 `locales/*.json`，代码中仅保留翻译键，支持通过 `config.json` 切换默认语言，并为未来多语言扩展留出 seam。

**Architecture:** 新增 `core/i18n.py` 作为翻译加载器，按 `locales/{locale}.json` 加载语言包；新增 `app.locale` 配置；所有玩家可见字符串通过 `i18n.get("minecraft.login.success")` 获取；保留 `zh-CN` 作为默认语言与 fallback。

**Tech Stack:** Python 3.11+, json, Pydantic Settings, pytest

## Global Constraints

- Python 3.11+，4 空格缩进，snake_case 模块/变量，PascalCase 类。
- 普通配置进入 `config.json` / `config.example.json`；密钥、密码只放 `.env`。
- 玩家会话隔离：若未来支持 per-player 语言，locale 必须存放在 `PlayerSession` 并随 `player_name` 传递，不得放在连接级状态。
- 每个任务必须有相邻测试，运行 `pytest tests/... -v` 验证。
- 提交遵循 Conventional Commits，例如 `feat(i18n): 添加国际化加载器`。
- 默认语言 `zh-CN` 必须始终存在，作为所有其他语言的 fallback。

- [ ] **Step 1: 创建 `core/i18n.py`**

```python
"""国际化（i18n）支持。

默认语言为 zh-CN。当请求的语言包不存在或某个键缺失时，回退到 zh-CN；
zh-CN 也缺失时返回键名本身，避免程序崩溃。
"""

import json
from pathlib import Path


LOCALES_DIR = Path(__file__).resolve().parent.parent / "locales"
DEFAULT_LOCALE = "zh-CN"


class I18n:
    """翻译加载器。"""

    def __init__(self, locale: str = DEFAULT_LOCALE, locales_dir: Path = LOCALES_DIR):
        self.locale = locale
        self.locales_dir = locales_dir
        self._messages: dict[str, object] = {}
        self._fallback_messages: dict[str, object] = {}
        self._load()

    def _load_file(self, locale: str) -> dict[str, object]:
        path = self.locales_dir / f"{locale}.json"
        if not path.exists():
            return {}
        try:
            return json.loads(path.read_text(encoding="utf-8"))
        except (json.JSONDecodeError, OSError):
            return {}

    def _load(self) -> None:
        self._messages = self._load_file(self.locale)
        if self.locale != DEFAULT_LOCALE:
            self._fallback_messages = self._load_file(DEFAULT_LOCALE)
        else:
            self._fallback_messages = {}

    def set_locale(self, locale: str) -> None:
        self.locale = locale
        self._load()

    def get(self, key: str, **kwargs) -> str:
        value = self._lookup(key, self._messages)
        if value is None:
            value = self._lookup(key, self._fallback_messages)
        if value is None:
            return key
        if kwargs:
            try:
                return value.format(**kwargs)
            except (KeyError, ValueError):
                return value
        return value

    @staticmethod
    def _lookup(key: str, messages: dict[str, object]) -> str | None:
        parts = key.split(".")
        current: object = messages
        for part in parts:
            if not isinstance(current, dict):
                return None
            current = current.get(part)
            if current is None:
                return None
        if isinstance(current, str):
            return current
        return None


_i18n = I18n()


def get_message(key: str, **kwargs) -> str:
    """获取翻译文本。

    Args:
        key: 点分翻译键，例如 "minecraft.login.success"。
        **kwargs: 用于格式化字符串的变量。

    Returns:
        翻译后的文本；若未找到则返回键名。
    """
    return _i18n.get(key, **kwargs)


def set_locale(locale: str) -> None:
    """切换全局默认语言。"""
    _i18n.set_locale(locale)


def get_locale() -> str:
    """获取当前全局语言。"""
    return _i18n.locale
```

```python
import json
from pathlib import Path

import pytest

from core.i18n import I18n, get_message, set_locale


def test_get_message_basic(tmp_path: Path):
    locales = tmp_path / "locales"
    locales.mkdir()
    (locales / "zh-CN.json").write_text(
        json.dumps({"minecraft": {"login": {"success": "登录成功！"}}}, ensure_ascii=False),
        encoding="utf-8",
    )

    i18n = I18n(locale="zh-CN", locales_dir=locales)
    assert i18n.get("minecraft.login.success") == "登录成功！"


def test_get_message_with_kwargs(tmp_path: Path):
    locales = tmp_path / "locales"
    locales.mkdir()
    (locales / "zh-CN.json").write_text(
        json.dumps({"agent": {"thinking": "AI正在为{player_name}思考"}}, ensure_ascii=False),
        encoding="utf-8",
    )

    i18n = I18n(locale="zh-CN", locales_dir=locales)
    assert i18n.get("agent.thinking", player_name="Steve") == "AI正在为Steve思考"


def test_missing_key_returns_key(tmp_path: Path):
    locales = tmp_path / "locales"
    locales.mkdir()
    (locales / "zh-CN.json").write_text("{}", encoding="utf-8")

    i18n = I18n(locale="zh-CN", locales_dir=locales)
    assert i18n.get("missing.key") == "missing.key"


def test_fallback_to_default_locale(tmp_path: Path):
    locales = tmp_path / "locales"
    locales.mkdir()
    (locales / "zh-CN.json").write_text(
        json.dumps({"common": {"hello": "你好"}}, ensure_ascii=False),
        encoding="utf-8",
    )
    (locales / "en-US.json").write_text(
        json.dumps({"common": {"bye": "Bye"}}, ensure_ascii=False),
        encoding="utf-8",
    )

    i18n = I18n(locale="en-US", locales_dir=locales)
    assert i18n.get("common.hello") == "你好"  # fallback
    assert i18n.get("common.bye") == "Bye"


def test_set_locale():
    set_locale("zh-CN")
    assert get_locale() == "zh-CN"
```
